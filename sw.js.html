const cacheName = 'gaylingo-pwa-v1'; const filesToCache = [ '/', '/index.html', '/manifest.json', '/sw.js' ]; // Install event self.addEventListener('install', e => { e.waitUntil( caches.open(cacheName).then(cache => cache.addAll(filesToCache)) ); }); // Fetch event self.addEventListener('fetch', e => { e.respondWith( caches.match(e.request).then(response => response || fetch(e.request)) ); });
